PŘEDNÁŠKA-26.2.26
SELECT pd.OS_CIS, pd.JMENO, pd.PLAT, nd.PLAT AS nadr_plat
FROM ZAM pd 
INNER JOIN ZAM nd ON pd.NADR = nd.OS_CIS
WHERE pd.PLAT > nd.PLAT
ORDER BY pd.OS_CIS;
